def Layer4_firewall (data:str):
    msg = ""
    y = 0
    x = data[0:15]
    for i in range(len(x)):
        y += int(x[14-i])*(2**i)
    if y == 80 or y == 22123:
        error="Allowed"
    else:
        error="Filtered"
        msg = x[15:]

    return error, msg
